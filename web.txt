        function handleNo() {
            const noButton = document.getElementById('noButton');
            const yesButton = document.getElementById('yesButton');
            // Make the No button move away (adjust for mobile)
            const maxMovement = window.innerWidth < 768 ? 100 : 200;
            const randomX = Math.random() * maxMovement - (maxMovement / 2);
            const randomY = Math.random() * maxMovement - (maxMovement / 2);
            noButton.style.transform = `translate(${randomX}px, ${randomY}px)`;
            noButton.style.transition = 'transform 0.5s ease';
            // Grow the Yes button
            const currentScale = parseFloat(yesButton.style.transform?.replace('scale(', '').replace(')', '')) || 1;
            const newScale = currentScale + 0.2;
            yesButton.style.transform = `scale(${newScale})`;
            yesButton.style.transition = 'transform 0.5s ease';
        }
